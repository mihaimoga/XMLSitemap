inline_or

The inline_or function creates an inline_or_executor object for an executor or execution context.

    typename Executor>
inline_or_executor< Executor > inline_or(
    const Executor & ex,
    constraint_t< is_executor< Executor >::value||execution::is_executor< Executor >::value >  = 0);
